import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner k = new Scanner(System.in);
        int l1 = k.nextInt();
        int l2 = k.nextInt();
        if (l1 % l2 == 0 || l2 % l1 == 0) {
            System.out.print("TAK");
        } else {
            System.out.print("NIE");
        }
        k.close();
    }
}
